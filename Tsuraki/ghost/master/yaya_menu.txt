OpenMenu
{
	//メニューに表示される、途中と最後の空行はわざと入れています。
	<<"
	りりす０５：メニューです。
	＠半分メニュー：なにか話して｜Menu_AITALK
	＠半分メニュー【STRLEN(SHIORI3FW.LastAITalk) > 0】：さっきの話は？｜Menu_LASTTALK
	＠半分メニュー：しゃべり頻度を変えて｜Menu_TALKINTERVAL
	＠半分メニュー：名前を覚えて｜Menu_TELLNAME
	＠半分メニュー：話しかけてみる｜Menu_USERCOMSTART
	＠半分メニュー：他のゴーストと話す｜Menu_GHOSTCOMSTART
	＠改行多めメニュー：あやりりすEXについて｜Menu_ABOUTAYALILITH
	＠メニュー：AIについて｜Menu_ABOUTAYA
	＠改行多めメニュー：なんでもない｜Menu_CANCEL
	">>
}

//---- 何か話して

{
	//---- 明示的にAIトーク開始を指示
	"\![raise,OnAITalk]"
}

//---- さっきの話は
Menu_LASTTALK
{
	//前のトークになにかある(＝文字列の長さが0ではない）時はそれを喋る
	if ( STRLEN(SHIORI3FW.LastAITalk) > 0 ) {
		SHIORI3FW.LastAITalk
	}
	else {
		<<"
		りりす０４：まだなにも喋ってないよ……
		">>
	}
}
